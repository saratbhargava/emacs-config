** ansi-term

Setting default shell to bash

#+BEGIN_SRC emacs-lisp
(defvar my-shell "/bin/bash")
(defadvice ansi-term (before force-bash)
  (interactive(list my-shell)))
(ad-activate 'ansi-term)
#+END_SRC

#+BEGIN_SRC emacs-lisp

  )
#+END_SRC

** keyboard keys mapping to suit emacs

#+BEGIN_SRC emacs-lisp
(when (eq system-type 'darwin)
  (setq mac-option-modifier 'meta) ;; alt is chaged to meta, as i didnt see any use for alt in emacs
  (setq mac-command-modifier 'meta)
  (setq mac-function-modifier 'control) ;; default function is none, so i changed to control for now.
  (global-set-key [kp-delete] 'delete-char)
  )
#+END_SRC
